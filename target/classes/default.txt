#Usage of MHR commands.
Cmd_Usage_Help=§6/mhr - コマンド一覧を表示します。
Cmd_Usage_SetSpeed=§6/mhr setspeed <double:speed> - 馬の速度を設定します。
Cmd_Usage_SetJump=§6/mhr setjump <double:jump> - 馬の速度を設定します。
Cmd_Usage_Spawn=§6/mhr spawn <int:num> [double:x,y,z] - 馬をスポーンします。
Cmd_Usage_Despawn=§6/mhr despawn [int:id] - 馬をデスポーンをします。
Cmd_Usage_Teleport=§6/mhr tp <int:id> [String:player] [boolean:flag] - 馬をテレポートします。
Cmd_Usage_TeleportLoc=§6/mhr tploc <int:id> <double:x,y,z> [double:yaw, pitch] [boolean:flag] - 馬を指定した座標にてレポートします。
Cmd_Usage_AllTeleport=§6/mhr alltp <double: x,y,z> [String:player] [boolean:flag] - すべての馬をテレポートします。
Cmd_Usage_AllTeleportLoc=§6/mhr alltploc <double: x,y,z> [double: yaw,pitch] [boolean: flag] - すべての馬を指定した座標にテレポートします。

#Usage of MHRC commands.
Cmd_Usage_Course.SetUsingCourse=§6/mhr c_usingcourse <String:course> - 使用するコースを指定します。
Cmd_Usage_Course.Add=§6/mhr c_add <String:CourseName> - コースを追加します。
Cmd_Usage_Course.Remove=§6/mhr c_remove <String:CourseName> - コースを削除します。
Cmd_Usage_Course.AddPoint=§6/mhr c_addpoint [int:index] - チェックポイントを追加します。
Cmd_Usage_Course.RemovePoint=§6/mhr c_rmpoint [int:index] - チェックポイントを消去します。
Cmd_Usage_Course.SetAngle=§6/mhr c_setangle <int:index> <int:angle> - チェックポイントが保持する角度を指定します。
Cmd_Usage_Course.SetOneLapIndex=§6/mhr c_setonelapindex <int:onelapindex> [String:course] - 1周に必要なチェックポイントの通過数を設定します。
Cmd_Usage_Course.SetLap=§6/mhr c_setlap <int:lap> [String:course] - ラップ数を設定します。
Cmd_Usage_Course.Start=§6/mhr c_start - レースを開始します。
